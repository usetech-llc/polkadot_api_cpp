56 Hasher CPolkaApi::getFuncHasher(unique_ptr<Metadata> &meta, const string &moduleName, const string &funcName) {
57  Hasher hasher = XXHASH;
58 
59  if (meta->metadataV0) {
60  hasher = XXHASH;
61  } else {
62  // Find the module index in metadata
63  int moduleIndex = getModuleIndex(meta, moduleName, false);
64 
65  // Find function by name in module and get its hasher
66  string hasherStr = "";
67  if (moduleIndex >= 0) {
68  int methodIndex = getStorageMethodIndex(meta, moduleIndex, funcName);
69  if (methodIndex > 0) {
70  if (meta->metadataV4)
71  hasherStr = meta->metadataV4->module[moduleIndex]->storage[methodIndex]->type.hasher;
72  else if (meta->metadataV5)
73  hasherStr = meta->metadataV5->module[moduleIndex]->storage[methodIndex]->type.hasher;
74  else if (meta->metadataV6)
75  hasherStr = meta->metadataV6->module[moduleIndex]->storage[methodIndex]->type.hasher;
76  }
77  }
78 
79  // Parse hasher name
80  if (hasherStr == "Blake2_256") {
81  hasher = BLAKE2;
82  }
83  }
84  return hasher;
85 }

87 int CPolkaApi::getModuleIndex(unique_ptr<Metadata> &meta, const string &moduleName, bool skipZeroCalls) {
88 
89  // Find the module index in metadata
90  int moduleIndex = -1;
91  int zeroMethodModuleCount = 0;
92  string moduleNameLower = moduleName;
93  std::transform(moduleNameLower.begin(), moduleNameLower.end(), moduleNameLower.begin(), easytolower);
94 
95  for (int i = 0; i < COLLECTION_SIZE; ++i) {
96  string name("");
97  if (meta->metadataV0 && meta->metadataV0->module[i]) {
98  name = meta->metadataV0->module[i]->prefix;
99  } else if (meta->metadataV4 && meta->metadataV4->module[i]) {
100  name = meta->metadataV4->module[i]->name;
101  } else if (meta->metadataV5 && meta->metadataV5->module[i]) {
102  name = meta->metadataV5->module[i]->name;
103  } else if (meta->metadataV6 && meta->metadataV6->module[i]) {
104  name = meta->metadataV6->module[i]->name;
105  }
106 
107  if (name.length()) {
108  std::transform(name.begin(), name.end(), name.begin(), easytolower);
109  if (!hasMethods(meta, i))
110  zeroMethodModuleCount++;
111 
112  if (moduleNameLower == name) {
113  moduleIndex = i;
114  if (skipZeroCalls) {
115  moduleIndex -= zeroMethodModuleCount;
116  }
117  break;
118  }
119  }
120  }
121 
122  return moduleIndex;
123 }

581 unsigned long CPolkaApi::getAccountNonce(string address) {
582  // Subscribe to account nonce updates and immediately unsubscribe after response is received
583  // This pattern is borrowed from POC-3 UI
584  bool done = false;
585  unsigned long result = 0;
586  subscribeAccountNonce(address, [&](unsigned long nonce) {
587  result = nonce;
588  done = true;
589  });
590 
591  while (!done)
592  usleep(10000);
593 
594  unsubscribeAccountNonce(address);
595 
596  return result;

747 void CPolkaApi::handleWsMessage(const int subscriptionId, const Json &message) {
748 
749  // TODO: DOT-55, fix with proper producer-consumer
750  int count = 0;
751  while (count < 10) {
752  // Handle Block subscriptions
753  if (_blockNumberSubscriptionId == subscriptionId) {
754  _blockNumberSubscriber(fromHex<long long>(message["number"].string_value()));
755  return;
756  }
757 
758  // Handle Account nonce subscriptions
759  for (auto const &sid : _nonceSubscriptionIds) {
760  if (sid.second == subscriptionId) {
761  _nonceSubscribers[sid.first](fromHex<unsigned long>(message["changes"][0][1].string_value(), false));
762  return;
763  }
764  }
765 
766  // Handle Balance subscriptions
767  for (auto const &sid : _balanceSubscriptionIds) {
768  if (sid.second == subscriptionId) {
769  _balanceSubscribers[sid.first](fromHex<uint128>(message["changes"][0][1].string_value(), false));
770  return;
771  }
772  }
773 
774  // Handle transaction completion subscriptions
775  if (_transactionCompletionSubscriptionId == subscriptionId) {
776  if (message.dump().find("ready") != std::string::npos)
777  _transactionCompletionSubscriber(string("ready"));
778  else if (message.dump().find("finalized") != std::string::npos) {
779  _transactionCompletionSubscriber(string("finalized"));
780 
781  // There is no need to unsubscribe, just reset variabled
782  _transactionCompletionSubscriber = nullptr;
783  _transactionCompletionSubscriptionId = 0;
784  }
785  return;
786  }
787 
788  // Handle Era and Session subscription
789  if (_eraAndSessionSubscriptionId == subscriptionId && _bestBlockNum != -1) {
790  auto lastLengthChange = fromHex<long long>(message["changes"][0][1].string_value(), false);
791  auto sessionLength = fromHex<long long>(message["changes"][1][1].string_value(), false);
792  auto currentEra = fromHex<long long>(message["changes"][2][1].string_value(), false);
793  auto sessionsPerEra = fromHex<long long>(message["changes"][3][1].string_value(), false);
794  auto currentIndexSubcription = fromHex<long long>(message["changes"][4][1].string_value(), false);
795 
796  auto sessionProgress = (_bestBlockNum - lastLengthChange + sessionLength) % sessionLength;
797  auto eraProgress = currentIndexSubcription % sessionsPerEra * sessionLength + sessionProgress;
798 
799  Era era;
800  era.currentEra = currentEra;
801  era.eraProgress = eraProgress;
802  Session session;
803  session.sessionIndex = currentIndexSubcription;
804  session.lastLengthChange = lastLengthChange;
805  session.sessionLength = sessionLength;
806  session.sessionProgress = sessionProgress;
807 
808  _eraAndSessionSubscriber(era, session);
809  }
810 
811  // Handle finalized head subscription
812  if (_finalizedBlockSubscriptionId == subscriptionId) {
813  BlockHeader blockHeader;
814  decodeBlockHeader(&blockHeader, message);
815  _finalizedBlockSubscriber(blockHeader);
816  return;
817  }
818 
819  // Handle runtime version subscription
820  if (_runtimeVersionSubscriptionId == subscriptionId) {
821  auto rtvPtr = createRuntimeVersion(message);
822  RuntimeVersion *rtv = rtvPtr.release();
823  _runtimeVersionSubscriber(*rtv);
824  delete rtv;
825  return;
826  }
827 
828  // Handle storage subscriptions
829  for (auto const &sid : _storageSubscriptionIds) {
830  if (sid.second == subscriptionId) {
831  _storageSubscribers[sid.first](message["changes"][0][1].string_value());
832  return;
833  }
834  }
835 
836  usleep(100000);
837  count++;
838  }
839 }

1010 void CPolkaApi::signAndSendTransfer(string sender, string privateKey, string recipient, uint128 amount,
1011  std::function<void(string)> callback) {
1012 
1013  _logger->info("=== Starting a Transfer Extrinsic ===");
1014 
1015  // Get account Nonce
1016  unsigned long nonce = getAccountNonce(sender);
1017  _logger->info(string("sender nonce: ") + to_string(nonce));
1018 
1019  // Format transaction
1020  TransferExtrinsic te;
1021  memset(&te, 0, sizeof(te));
1022  te.method.moduleIndex = _protocolPrm.BalanceModuleIndex;
1023  te.method.methodIndex = _protocolPrm.TransferMethodIndex;
1024  auto recipientPK = AddressUtils::getPublicKeyFromAddr(recipient);
1025  memcpy(te.method.receiverPublicKey, recipientPK.bytes, PUBLIC_KEY_LENGTH);
1026  te.method.amount = amount;
1027  te.signature.version = SIGNATURE_VERSION;
1028  auto senderPK = AddressUtils::getPublicKeyFromAddr(sender);
1029  memcpy(te.signature.signerPublicKey, senderPK.bytes, PUBLIC_KEY_LENGTH);
1030  te.signature.nonce = nonce;
1031  te.signature.era = IMMORTAL_ERA;
1032 
1033  // Format signature payload
1034  SignaturePayload sp;
1035  sp.nonce = nonce;
1036  uint8_t methodBytes[MAX_METHOD_BYTES_SZ];
1037  sp.methodBytesLength = te.serializeMethodBinary(methodBytes);
1038  sp.methodBytes = methodBytes;
1039  sp.era = IMMORTAL_ERA;
1040  memcpy(sp.authoringBlockHash, _protocolPrm.GenesisBlockHash, BLOCK_HASH_SIZE);
1041 
1042  // Serialize and Sign payload
1043  uint8_t signaturePayloadBytes[MAX_METHOD_BYTES_SZ];
1044  long payloadLength = sp.serializeBinary(signaturePayloadBytes);
1045 
1046  vector<uint8_t> secretKeyVec = fromHex<vector<uint8_t>>(privateKey);
1047  uint8_t sig[SR25519_SIGNATURE_SIZE] = {0};
1048  sr25519_sign(sig, te.signature.signerPublicKey, secretKeyVec.data(), signaturePayloadBytes, payloadLength);
1049 
1050  // Copy signature bytes to transaction
1051  memcpy(te.signature.sr25519Signature, sig, SR25519_SIGNATURE_SIZE);
1052 
1053  // Serialize and send transaction
1054  uint8_t teBytes[MAX_METHOD_BYTES_SZ];
1055  long teByteLength = te.serializeBinary(teBytes);
1056  string teStr("0x");
1057  for (int i = 0; i < teByteLength; ++i) {
1058  char b[3] = {0};
1059  sprintf(b, "%02X", teBytes[i]);
1060  teStr += b;
1061  }
1062 
1063  Json query = Json::object{{"method", "author_submitAndWatchExtrinsic"}, {"params", Json::array{teStr}}};
1064 
1065  // Send == Subscribe callback to completion
1066  _transactionCompletionSubscriber = callback;
1067  _transactionCompletionSubscriptionId = _jsonRpc->subscribeWs(query, this);
1068 }
