45 bool verify_subject_alternative_name(const char *hostname, X509 *cert) {
46  return true;
47 
48  STACK_OF(GENERAL_NAME) *san_names = NULL;
49 
50  san_names = (STACK_OF(GENERAL_NAME) *)X509_get_ext_d2i(cert, NID_subject_alt_name, NULL, NULL);
51  if (san_names == NULL) {
52  return false;
53  }
54 
55  int san_names_count = sk_GENERAL_NAME_num(san_names);
56 
57  bool result = false;
58 
59  for (int i = 0; i < san_names_count; i++) {
60  const GENERAL_NAME *current_name = sk_GENERAL_NAME_value(san_names, i);
61 
62  if (current_name->type != GEN_DNS) {
63  continue;
64  }
65 
66  char *dns_name = (char *)ASN1_STRING_data(current_name->d.dNSName);
67 
68  // Make sure there isn't an embedded NUL character in the DNS name
69  if (ASN1_STRING_length(current_name->d.dNSName) != (int)strlen(dns_name)) {
70  break;
71  }
72  // Compare expected hostname with the CN
73  result = (strcasecmp(hostname, dns_name) == 0);
74  }
75  sk_GENERAL_NAME_pop_free(san_names, GENERAL_NAME_free);
76 
77  return result;
78 }

81 bool verify_common_name(const char *hostname, X509 *cert) {
82  return true;
83 
84  // Find the position of the CN field in the Subject field of the certificate
85  int common_name_loc = X509_NAME_get_index_by_NID(X509_get_subject_name(cert), NID_commonName, -1);
86  if (common_name_loc < 0) {
87  return false;
88  }
89 
90  // Extract the CN field
91  X509_NAME_ENTRY *common_name_entry = X509_NAME_get_entry(X509_get_subject_name(cert), common_name_loc);
92  if (common_name_entry == NULL) {
93  return false;
94  }
95 
96  // Convert the CN field to a C string
97  ASN1_STRING *common_name_asn1 = X509_NAME_ENTRY_get_data(common_name_entry);
98  if (common_name_asn1 == NULL) {
99  return false;
100  }
101 
102  char *common_name_str = (char *)ASN1_STRING_data(common_name_asn1);
103 
104  // Make sure there isn't an embedded NUL character in the CN
105  if (ASN1_STRING_length(common_name_asn1) != (int)strlen(common_name_str)) {
106  return false;
107  }
108 
109  // Compare expected hostname with the CN
110  return (strcasecmp(hostname, common_name_str) == 0);
111 }

119 bool verify_certificate(const char *hostname, bool preverified, boost::asio::ssl::verify_context &ctx) {
120  // The verify callback can be used to check whether the certificate that is
121  // being presented is valid for the peer. For example, RFC 2818 describes
122  // the steps involved in doing this for HTTPS. Consult the OpenSSL
123  // documentation for more details. Note that the callback is called once
124  // for each certificate in the certificate chain, starting from the root
125  // certificate authority.
126 
127  // Retrieve the depth of the current cert in the chain. 0 indicates the
128  // actual server cert, upon which we will perform extra validation
129  // (specifically, ensuring that the hostname matches. For other certs we
130  // will use the 'preverified' flag from Asio, which incorporates a number of
131  // non-implementation specific OpenSSL checking, such as the formatting of
132  // certs and the trusted status based on the CA certs we imported earlier.
133  int depth = X509_STORE_CTX_get_error_depth(ctx.native_handle());
134 
135  // if we are on the final cert and everything else checks out, ensure that
136  // the hostname is present on the list of SANs or the common name (CN).
137  if (depth == 0 && preverified) {
138  X509 *cert = X509_STORE_CTX_get_current_cert(ctx.native_handle());
139 
140  if (verify_subject_alternative_name(hostname, cert)) {
141  return true;
142  } else if (verify_common_name(hostname, cert)) {
143  return true;
144  } else {
145  return false;
146  }
147  }
148 
149  if (!preverified) {
150  throw websocketpp::exception(X509_verify_cert_error_string(X509_STORE_CTX_get_error(ctx.native_handle())));
151  }
152 
153  return preverified;
154 }

189 int CWebSocketClient::connect() {
190 
191  std::string uri = _nodeUrl;
192 
193  try {
194  // Disable websocket++ logging, we have our own
195  _c.set_access_channels(websocketpp::log::alevel::none);
196  _c.set_error_channels(websocketpp::log::elevel::none);
197 
198  _c.set_open_handler(bind(&on_open, &_c, ::_1));
199 
200  // Initialize ASIO
201  _c.init_asio();
202 
203  // Register our message handler
204  _c.set_message_handler(&on_message);
205  _c.set_tls_init_handler(bind(&on_tls_init, uri.c_str(), ::_1));
206 
207  websocketpp::lib::error_code ec;
208  _connection = _c.get_connection(uri, ec);
209  if (ec) {
210  _logger->error("Could not create connection: " + ec.message());
211  return PAPI_CANT_CONNECT;
212  }
213 
214  // Note that connect here only requests a connection. No network messages are
215  // exchanged until the event loop starts running in the next line.
216  _c.connect(_connection);
217 
218  _logger->info(string("Connecting to ") + uri);
219 
220  // Start the ASIO io_service run loop
221  // this will cause a single connection to be made to the server. c.run()
222  std::unique_lock<std::mutex> connectionWaitLock(_connectionMtx);
223  _connectedThread = new thread(&CWebSocketClient::runWsMessages, this);
224  _healthThread = new thread(&CWebSocketClient::health, this);
225 
226  // Wait for connection
227  _connectionCV.wait_for(connectionWaitLock, ConnectionTimeout);
228  if (_connected) {
229  _logger->info("Connection established");
230  } else {
231  _connectedThread->join();
232  _healthThread->join();
233  delete _connectedThread;
234  delete _healthThread;
235  _connectedThread = nullptr;
236  _healthThread = nullptr;
237  _logger->error("Connection failed");
238  return PAPI_CANT_CONNECT;
239  }
240 
241  } catch (websocketpp::exception const &e) {
242  _logger->error(string("Could not create connection: ") + e.what());
243  return PAPI_CANT_CONNECT;
244  }
245 
246  return PAPI_OK;
247 }
